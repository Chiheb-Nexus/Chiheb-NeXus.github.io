    Parity: Vos premiers pas avec Ethereum

    Introduction

    Parity est un client fiable et performant,permettant de se connecter au réseau Ethereum sous licence GPLv3. Parity, par défaut, permet un accès rapide à un portefeuille d’Ethereum type Parity Wallet. L’outil permet de:

    • Créer et gérer les comptes Ethereum
    • Gérer les ethers et les jetons Ethereum
    • Créer et enregistrer les jetons
    • Créer, déployer et interagir avec les smarts contracts
    • etc.

    Par défaut, Parity démarre un serveur JSONRPC sur 127.0.0.1:8545 et un serveur de websockets sur 127.0.0.1:8546 (ceci est configurable) et supporte plusieurs autres type d’APIs.

    Installation:

    Il suffit d’aller sur https://www.parity.io et télécharger le binaire suivant votre OS.

    Par exemple, si vous êtes sous une distribution basée sur Debian, c’est simple, téléchargez le binaire, ouvrez une terminale et lancez :

    sudo dpkg -i parity_1.8.7_amd64.deb

    Puis, assurez-vous que Parity a été bien installé, puis lancez :

    parity --version

    Si vous vous êtes bien débrouillé, votre écran devrait vous montrer ça :

    Créons notre Smart Contract :

    example_contract.sol

    pragma solidity ^0.4.0;

/* 
	Un contract abstrait où on déclare les fonctions et les évènements à utiliser.
	NB: Un contract abstrait ne peut pas etre compilé. Cependant ce type de contract
		peuvent etre utilisé comme une base pour d'autre contracts
*/

contract AbstractContract {
	/*
		LogUserGetToken: Un log dans la EVM quand un utiliseur reçoit ses tokens
		LogUserTokenBalance: Un log dans le EVM qui permet de tracer le retour 
						  de la fonction userTokenBalance
		LogTransferToken: Un log dans la EVM qui permet le traçage de l'envoi 
						  des tokens d'une addresse vers une autre
		LogWithdrawEtherToOwner: Un log dans le EVM qui permet le traçage du withdraw
						  vers l'addresse du propriétaire du contract

		userTokenBalance: Permet de connaitre la balance en Token d'une addresse
		transferToken: Permet le transfert des tokens d'une addresse à une autre
		withdrawEtherToOwner: Permet de vider l'addresse du contract des ethers
						  et les envoyer vers l'addresse du propriétaire du contract
	*/
	event LogUserGetToken(address _user, uint _amount, string _tokenName);
	event LogUserTokenBalance(address _user, uint _amount, string _tokenName);
	event LogTransferToken(address _from, address _to, uint _amount, string _tokenName);
	event LogWithdrawEtherToOwner(address _contractAdd, address _owner, uint _amount, string _tokenName);
	function userTokenBalance(address _user) public returns (uint);
	function transferToken(address _from, address _to, uint amount) public;
	function withdrawEtherToOwner() public;
}

// MyTokenContract hérite les fonctions du contract AbstractContract
contract MyTokenContract is AbstractContract {

	// address du propriétaire du contract
	address public owner;
	// Nom du jeton
	string public tokenName;
	// Le maximum des jetons qui peuvent etre générés
	// en se basant à 18 décimals le maxSupply = 42 unités
	uint constant public maxSupply = 42 * 10 ** 18;
	// Logger les jetons restants
	uint public currentSupply;
	// Un dictionnaire où on peut savoir la balance de chaque addresse avec ses jetons
	mapping (address => uint) public balances;

	function MyTokenContract() public {
		// Initialiser le contract
		// Cette fonction sera exécutée une seule fois dans la vie du contract
		currentSupply = maxSupply;
		tokenName = 'MyTokenERC20';
		// Le propriétaire du contract est celui qui le déploie
		// msg.sender = address de celui qui a déployé le contract
		owner = msg.sender;
	}

	// Fallback
	// Cette fonction est un "payable" c'est à dire elle accepte les dépots en ethers
	function () public payable {
		// 1 ether = 1 MyTokenERC20
		// Et il faut pas que les ether reçus dépasse ce qui reste de jetons
		// Car on va pas créer des nouveaux jetons dans ce smart contract
		require(currentSupply >= msg.value);
		currentSupply -= msg.value;
		// Incrémenter la balance de celui qui a envoyé les ethers en jetons
		// PS: si le key n'existe pas, il sera ajouté automatiquement
		// et sa valeur par défaut sera égale à 0
		// du coup on peut l'incrémenter sans causer aucune erreur
		// dans le déroulement du contract
		balances[msg.sender] += msg.value;
		// Un log
		LogUserGetToken(msg.sender, msg.value, tokenName);

	}

	function userTokenBalance(address _user) public returns (uint balance) {
		balance = balances[_user];
		LogUserTokenBalance(_user, balance, tokenName);
	}

	function transferToken(address _from, address _to, uint _amount) public {
		// Une addresse ne peut pas envoyer des tokens à elle meme
		require(_from != _to);
		// La balance de celui qui va envoyer les jetons doit etre supérieure à _amount
		require(balances[_from] >= _amount);
		balances[_from] -= _amount;
		balances[_to] += _amount;
		// Un log
		LogTransferToken(_from, _to, _amount, tokenName);
	}

	function withdrawEtherToOwner() public {
		// Cette fonction doit etre exécuté uniquement par le propriétaire du contract
		require(msg.sender == owner);
		// Calcul du GAS total de la transaction
		// GAS utilisé est 21000
		var gasConsumed = tx.gasprice * 21000;
		var finalAmount = this.balance - gasConsumed;
		// Transférer les ethers du smart contract vers l'address du propriétaire
		owner.transfer(finalAmount);
		LogWithdrawEtherToOwner(this, owner, finalAmount, 'ETH');

	}
}

    Ensuite, on envoie quelques ethers à l’adresse de notre smart contract et on lance la simulation:

    Le code Python et web3.py qui va communiquer avec le smart contract.

    interaction_with_contract.py

    import json
from web3 import Web3, RPCProvider
from web3.contract import ConciseContract

RPC_IP = '127.0.0.1'
RPC_PORT = '8545'

# read the contract informations then convert them into a Python dict
with open('contract_informations.json', 'r') as f:
	data = json.loads(f.read())

ABI = data.get('abi')
CONTARCT_ADDRESS = data.get('contract_address')

w3 = Web3(RPCProvider(RPC_IP, RPC_PORT))

# Read the contract in ConciseContract mode which is a Read only access
contract_instance = w3.eth.contract(ABI, CONTARCT_ADDRESS, ContractFactoryClass=ConciseContract)
addresses = ['0x00a329c0648769A73afAc7F9381E08FB43dBEA72', '0x00F2b2abFD149fe48F6f051D8AD99b8A100AeE58']

# Check smart contract balance in ethers
smart_contract_balance = w3.eth.getBalance(CONTARCT_ADDRESS)
print('Balance of the smart contract ({0}) : {1} ethers'.format(CONTARCT_ADDRESS, smart_contract_balance))

# Check balances
for addr in addresses:
	balance = contract_instance.userTokenBalance(addr)
	print('Balance of {0} = {1} (10 ** 18 unités)'.format(addr, balance))
	
# Send token from 0x00a329c0648769A73afAc7F9381E08FB43dBEA72 to 0x00F2b2abFD149fe48F6f051D8AD99b8A100AeE58
# PS: Vous devez confirmer la transaction dans l'interface de Parity !!!
tx_hash = contract_instance.transferToken(addresses[0], addresses[1], 1 * 10 ** 18, transact={'from': w3.eth.accounts[0], 'gasPrice': 21000})
print(tx_hash)

# Check balances
for addr in addresses:
	balance= contract_instance.userTokenBalance(addr)
	print('Balance of {0} = {1} (10 ** 18 unités)'.format(addr, balance))
	
# Withdraw all ethers to owner
# Vous devez Confirmer la transaction dans l'interface de Parity !!
tx_hash = contract_instance.withdrawEtherToOwner(transact={'from': w3.eth.accounts[0], 'gasPrice': 21000})
print(tx_hash)

# Check smart contract balance in ethers
smart_contract_balance = w3.eth.getBalance(CONTARCT_ADDRESS)
print('Balance of the smart contract ({0}) : {1} ethers'.format(CONTARCT_ADDRESS, smart_contract_balance))
